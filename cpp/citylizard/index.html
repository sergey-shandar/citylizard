CityLizard++

CityLizard++ is a set of general header-only C++ libraries.

1. Binary Literals

Header citylizard/binary/b.hpp.

Namespace citylizard::binary. 

namespace _detail
{
    template<unsigned int X>
    struct b
    {
        static unsigned int x = X;

        typedef b<...> _0000;
        typedef b<...> _0001;
        ...
        typedef b<...> _1111;
    };
}

typedef _detail::b<0x0> _0000;
typedef _detail::b<0x1> _0001;
...
typedef _detail::b<0xF> _1111;

1.1. Example

using namespace citylizard::binary;
unsigned int xE19F = _1110::_0001::_1001::_1111::x;

4. Intrusive Pointer and Reference

Namespace citylizard::intrusive.

4.1. Intrusive Pointer

template
struct ptr
{
    ptr() throw();
    ptr(ptr const &) throw();
    ptr(ptr &&) throw();
    ptr(ref const &) throw();

    ptr &operator=(ptr const &) throw();
    ptr &operator=(ptr &&) throw();
    ptr &operator=(ref const &) throw();

    void reset() throw();

    void swap(ptr &) throw();
    void swap(ref &) throw(...);

    bool operator==(ptr const &) const throw();
    bool operator==(ref const &) const throw();

    bool operator!=(ptr const &) const throw();
    bool operator!=(ref const &) const throw();

    T &operator*() const throw(...);
    T *operator->() const throw(...);

    bool bool_cast() const throw();

    template
    ptr up_cast() const throw();

    template
    ptr up_cast() && throw();
};
